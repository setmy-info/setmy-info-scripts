# PLACEHOLDER-BEGIN #
MESSAGE("-- cpp variables.txt")

# -Wall -Wextra -pedantic -Werror -Wfloat-conversion
ADD_COMPILE_OPTIONS(-Wall -Wextra -pedantic)
ADD_COMPILE_OPTIONS(-DPTHREADSAFE)

SET (CMAKE_CXX_STANDARD 17)
SET (CMAKE_CXX_STANDARD_REQUIRED ON)
SET (CMAKE_CXX_EXTENSIONS OFF)

IF(CMAKE_BUILD_TYPE STREQUAL "Debug")
    ADD_COMPILE_OPTIONS(-DDEBUG -D_DEBUG)
ENDIF()

IF(CMAKE_SYSTEM_NAME STREQUAL "FreeBSD")
    SET (EXAMPLE_LINKING            "-lexampleshared -pthread")
    SET (EXAMPLE_TEST_LINKING       "-lexampleshared -pthread")
ENDIF()

IF(CMAKE_SYSTEM_NAME STREQUAL "Linux" )
    SET (EXAMPLE_LINKING            "-lexampleshared -pthread -ldl")
    # -lboost_unit_test_framework
    SET (EXAMPLE_TEST_LINKING       "-lexampleshared -pthread -ldl")
ENDIF()

SET (CMAKE_C_FLAGS_RELEASE "${CMAKE_C_FLAGS_RELEASE} -s")
SET (CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -s")
SET (MEM_LEAK_TEST_COMMAND valgrind --error-exitcode=1 --leak-check=full --show-leak-kinds=all)

INCLUDE_DIRECTORIES("${MAIN_INCLUDES_PATH}")
INCLUDE_DIRECTORIES("${INCLUDE_OUTPUT_PATH}")
INCLUDE_DIRECTORIES("${TEST_INCLUDES_PATH}")

LINK_DIRECTORIES("${BUILD_PATH}/lib")

# PLACEHOLDER-END #
