function(INCLUDE_CMAKE_FILE)
    IF(EXISTS ${ARGV0})
        MESSAGE("INCLUDING: ${ARGV0}")
        INCLUDE(${ARGV0})
    ELSE()
        MESSAGE("NOT INCLUDING: ${ARGV0} - FILE DOESNT EXIST")
    ENDIF()
endfunction()

function(INCLUDE_DIRECTORIES_WINDOWS)
    IF (WIN32)
        INCLUDE_DIRECTORIES(${ARGV0})
    ENDIF()
endfunction()
